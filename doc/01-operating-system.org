#+begin_src shell-script :tangle ../profile/01-operating-system.sh
  ensure-platform() {
      case "$(kernel-type)" in
           ,*'linux'*)       PLATFORM='linux'    ;;
          ,*'darwin'*)       PLATFORM='osx'      ;;
             ,*'bsd'*)       PLATFORM='bsd'      ;;
            ,*'msys'*)       PLATFORM='msys'     ;;
          ,*'cygwin'*)       PLATFORM='cygwin'   ;;
                   ,*)       PLATFORM='unknown'  ;;
      esac

      export PLATFORM
  }
#+end_src


#+begin_src shell-script :tangle ../profile/01-operating-system.sh
  is-linux() {
      ensure-platform

      if [ "$PLATFORM" = "linux" ]; then
          return $exit_success
      else
          return $exit_failure
      fi
  }
#+end_src
